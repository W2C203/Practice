    function cuturl(url){
        var re = {};              //存返回值
        url = url.split("?")[1];  //取得？后面的字符串
        var get_thing = url.split("&");
        for(var i = 0; i < get_thing.length ; i++){
            var temp = get_thing[i].split("=");
            re[temp[0]] = temp[1];
        }
        return JSON.stringify(re);
    }
    var url="http://item.taobao.com/item.htm?a=1&b=2&c=&d=xxx&e";
    document.write("url:" + url + "   " + cuturl(url));
