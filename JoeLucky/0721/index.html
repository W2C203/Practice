默认转换“ <|>|&|"|' ” 
        String.prototype.escapeHtml= function (){
        return this.replace(/[<>&"']/g,function(x){
            switch(x){
                case "<": return "<";
                case ">": return ">";
                case "&": return "&";
                case '"': return "&apos"; //(IE不支持)
                case "'": return """;
            }
        });
    }
    document.getElementById("two").onclick = function(){
        var str= document.getElementById("one").value;
        if(!str) str= "<|>|&|\"|'";
        console.log("原字符串："+str);
        console.log("转换后："+str.escapeHtml());
    }
