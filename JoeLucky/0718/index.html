错误的做法：
    var isArray= function(arr) {
        return arr.constructor == Array;
    }
    //理解constructor!!!  constructor 属性返回对创建此对象的数组函数的引用 并且其值的可变性
    function newarray(){     return new Array(4);     }
    function newint(){       return 10;    }

    var aaa= new newarray();
    alert(aaa.constructor); // function Array() { [native code] }
    var bbb= new newint();
    alert(bbb.constructor); //  function newint(){ return 10; }

    //这时候如果我这么做：
    aaa.constructor=newarray;
    //毫无疑问 已经达不到预期效果
    alert(aaa.constructor);// function newarray(){return new Array(4);}


    正确的做法：
    var isArray = Function.isArray ||  //ECMAScript5 新增的方法
        function (obj){
            return typeof obj === 'object' && Object.prototype.toString.call(obj) === '[object Array]';
            //由于Array中对Object中toString方法进行了重写 那么通过call方法 对obj引用Object的方法 然后判断是否为数组对象
        }
